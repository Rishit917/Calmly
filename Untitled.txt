import streamlit as st
from groq import Groq

# 1. The Title
st.title("My AI Therapist")
st.write("Talk to me about anything.")

# 2. Connect the Brain (Paste your key from Step A here)
client = Groq(api_key="PASTE_YOUR_KEY_HERE")

# 3. Create the Chat Box
if user_text := st.chat_input("How are you feeling today?"):
    st.chat_message("user").write(user_text)

    # 4. Tell the AI how to act (The Personality)
    response = client.chat.completions.create(
        model="llama3-8b-8192",
        messages=[
            {"role": "system", "content": "You are a kind therapist. Ask questions to help the user."},
            {"role": "user", "content": user_text}
        ]
    )

    # 5. Show the Answer
    st.chat_message("assistant").write(response.choices[0].message.content)
